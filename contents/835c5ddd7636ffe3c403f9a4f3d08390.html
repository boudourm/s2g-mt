public void duplicateSubscribersFrom(Long idCup, Long idCup2, Long idOrg) throws MTDataBaseException, MTServiceException { log.info("duplicateSubscribersFrom"); boolean isSiteManager = organisationRepository.isOrganisationSiteManager(idOrg);

	CollUsageProduction collUsageProduction = collUsageProductionRepository.findCollUsageProductionById(idCup, idOrg, isSiteManager);
	if(collUsageProduction == null) {
		log.info("Couldn't find cup with id="+idCup);
		throw new MTServiceException(MTInternalCodeType.COLL_USAGE_PROD_NOT_EXISTS_ERROR);
	}
	CollUsageProduction collUsageProduction2 = collUsageProductionRepository.findCollUsageProductionById(idCup2, idOrg, isSiteManager);
	if(collUsageProduction2 == null) {
		log.info("Couldn't find cup with id="+idCup2);
		throw new MTServiceException(MTInternalCodeType.COLL_USAGE_PROD_NOT_EXISTS_ERROR);
	}

	List<CuSubscription> subs = new ArrayList<CuSubscription>();
	for (CuSubscription cs: collUsageProduction2.getCuSubscriptions()) {
		subs.add(new CuSubscription(new CuSubscriptionId(idCup, cs.getId().getIdPre()), collUsageProduction, cs.getPremises()));
	}

	cuSubscriptionRepository.insertMany(subs);

	collUsageProduction = collUsageProductionRepository.findCollUsageProductionById(idCup, idOrg, isSiteManager);
}
